### 2.3 状态管理 (Pinia)

```typescript
// stores/session.ts
import { defineStore } from 'pinia';
import { ref, computed } from 'vue';
import type { Session, Message } from '@/types/session';
import { chatService } from '@/services/chatService';

export const useSessionStore = defineStore('session', () => {
  // State
  const sessions = ref<Session[]>([]);
  const currentSessionId = ref<string | null>(null);
  const isLoading = ref(false);
  const error = ref<string | null>(null);

  // Getters
  const currentSession = computed(() =>
    sessions.value.find(s => s.id === currentSessionId.value)
  );

  // Actions
  async function fetchSessions() {
    isLoading.value = true;
    error.value = null;
    try {
      sessions.value = await chatService.getSessions();
    } catch (e) {
      error.value = (e as Error).message;
    } finally {
      isLoading.value = false;
    }
  }

  async function createSession() {
    const session = await chatService.createSession();
    sessions.value.unshift(session);
    currentSessionId.value = session.id;
    return session;
  }

  async function deleteSession(id: string) {
    await chatService.deleteSession(id);
    sessions.value = sessions.value.filter(s => s.id !== id);
    if (currentSessionId.value === id) {
      currentSessionId.value = sessions.value[0]?.id ?? null;
    }
  }

  function switchSession(id: string) {
    currentSessionId.value = id;
  }

  function addMessage(sessionId: string, message: Message) {
    const session = sessions.value.find(s => s.id === sessionId);
    if (session) {
      session.messages.push(message);
    }
  }

  function updateMessage(sessionId: string, messageId: string, content: string) {
    const session = sessions.value.find(s => s.id === sessionId);
    if (session) {
      const msg = session.messages.find(m => m.id === messageId);
      if (msg) msg.content = content;
    }
  }

  return {
    // State
    sessions,
    currentSessionId,
    isLoading,
    error,
    // Getters
    currentSession,
    // Actions
    fetchSessions,
    createSession,
    deleteSession,
    switchSession,
    addMessage,
    updateMessage,
  };
});
```

## 4. 数据库设计 (Supabase)

### 4.1 表结构概览

所有表均包含统一审计字段：
- `created_by` (UUID → auth.users) — 创建人
- `created_at` (TIMESTAMPTZ) — 创建时间
- `updated_by` (UUID → auth.users) — 更新人
- `updated_at` (TIMESTAMPTZ) — 更新时间

通过触发器自动填充：INSERT 时 `set_created_by()` 设置 `created_by`/`updated_by`，UPDATE 时 `set_updated_by()` 刷新 `updated_by`/`updated_at`。

### 4.2 表清单

#### settings — 用户配置
| 字段 | 类型 | 说明 |
|------|------|------|
| id | UUID PK | 主键 |
| user_id | UUID FK → auth.users | 用户（唯一约束） |
| api_key | TEXT | 用户 API Key |
| model | TEXT | 模型（默认 gpt-4o） |
| temperature | FLOAT | 温度（默认 0.7） |
| max_tokens | INT | 最大 Token（默认 2000） |

RLS：`auth.uid() = user_id`

#### sessions — 会话
| 字段 | 类型 | 说明 |
|------|------|------|
| id | UUID PK | 主键 |
| user_id | UUID FK → auth.users | 用户 |
| title | TEXT | 标题（默认"新对话"） |

RLS：`auth.uid() = user_id`

#### messages — 消息
| 字段 | 类型 | 说明 |
|------|------|------|
| id | UUID PK | 主键 |
| session_id | UUID FK → sessions | 所属会话（级联删除） |
| role | TEXT | 'user' 或 'assistant' |
| content | TEXT | 消息内容 |
| model | TEXT | AI 模型（仅 assistant 有） |

RLS：通过子查询验证 `sessions.user_id = auth.uid()`

## 8. 注意事项

### 8.1 API Key 安全

- 用户配置的 API Key 必须加密存储
- 后端解密后直接用于 AI API 调用，不返回给前端
- 建议使用 Supabase Vault 进行敏感数据存储

### 8.2 流式响应兼容

- 使用 Server-Sent Events (SSE) 实现流式响应
- 前端使用 `fetch` + `ReadableStream` 解析
- 确保代理服务器（如 Nginx）支持 SSE

### 8.3 错误处理

- 前端对所有 API 调用进行错误捕获
- 后端使用 Pydantic 进行请求验证
- 错误响应格式统一：`{ "error": "错误信息" }`

### 8.4 架构说明

- 前端**不直接访问 Supabase**，所有数据通过 FastAPI 后端中转，保障 Supabase `service_role` 密钥安全
- 前端 API 调用有 localStorage 回退机制：后端不可用时自动降级到本地存储，便于独立开发 UI
- 认证方式：后端通过 Supabase Auth SDK 处理注册/登录，返回 JWT token 给前端
